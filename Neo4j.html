Some examples below:

G = nx.MultiDiGraph()
nodes = list(result.graph()._nodes.values())
for node in nodes:
    G.add_node(node.id, labels=node._labels)
edges = list(result.graph()._relationships.values())
for edge in edges:
    G.add_edge(edge.start_node.id, edge.end_node.id)
